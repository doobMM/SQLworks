부모의 형질을 모두 가지는 대장균 찾기

/* ME */
SELECT D1.ID, D1.GENOTYPE, D2.GENOTYPE AS PARENT_GENOTYPE
  FROM ECOLI_DATA D1
  JOIN ECOLI_DATA D2 ON D1.PARENT_ID = D2.ID
 WHERE D2.GENOTYPE & D1.GENOTYPE = 1
 ORDER BY D1.ID

/* GPT */
SELECT D1.ID, D1.GENOTYPE, D2.GENOTYPE AS PARENT_GENOTYPE
  FROM ECOLI_DATA D1
  JOIN ECOLI_DATA D2 ON D1.PARENT_ID = D2.ID
 WHERE D2.GENOTYPE & D1.GENOTYPE = D2.GENOTYPE
 ORDER BY D1.ID
