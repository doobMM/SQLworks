/* ME DRAFT */
SELECT ID, COUNT() AS CHILD_COUNT
  FROM ECOLI_DATA D1
  JOIN ECOLI_DATA D2 ON D1.ID = D2.PARENT_ID
 ORDER BY ID ASC

/* GPT */
SELECT D1.ID, COUNT(D2.ID) AS CHILD_COUNT
  FROM ECOLI_DATA D1
  LEFT JOIN ECOLI_DATA D2 ON D1.ID = D2.PARENT_ID
 GROUP BY D1.ID
 ORDER BY D1.ID ASC;

/* LEFT JOIN 사용: 일부 부모 개체가 자식이 없을 수 있으므로, 부모 개체도 모두 포함시키기 위해 LEFT JOIN을 사용합니다.

GROUP BY: 부모 ID별로 자식 개체 수를 집계하려면 GROUP BY D1.ID가 필요합니다.*/
