분기별 분화된 대장균의 개체 수 구하기

/*ME*/
SELECT CONCAT(QUARTER(DIFFERENTIATION_DATE), 'Q') AS QUARTER,
       COUNT(ID) AS ECOLI_COUNT
  FROM ECOLI_DATA
 GROUP BY QUARTER(DIFFERENTIATION_DATE)
 ORDER BY QUARTER(DIFFERENTIATION_DATE);

-- 오류가 발생한다. 

/* WITH GPT */
SELECT CONCAT(EXTRACT(QUARTER FROM DIFFERENTIATION_DATE), 'Q') AS QUARTER,
       COUNT(ID) AS ECOLI_COUNT
  FROM ECOLI_DATA
 GROUP BY CONCAT(EXTRACT(QUARTER FROM DIFFERENTIATION_DATE), 'Q')
 ORDER BY QUARTER;
